def read2DPoints(path):
    x, y = [], []
    with open(path, 'r') as file:
        lines = file.read().splitlines()
        for line in lines:
            x1, y1 = line.split()
            x.append(float(x1))
            y.append(float(y1))
    return np.array(x), np.array(y)

def normalize(X):
    return (X - min(X)) / (max(X) - min(X))
